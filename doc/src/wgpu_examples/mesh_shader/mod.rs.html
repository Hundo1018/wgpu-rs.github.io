2fn compile_wgsl(device: &wgpu::Device) -> wgpu::ShaderModule {
3    // Workgroup memory zero initialization can be expensive for mesh shaders
4    unsafe {
5        device.create_shader_module_trusted(
6            wgpu::ShaderModuleDescriptor {
7                label: None,
8                source: wgpu::ShaderSource::Wgsl(include_str!("shader.wgsl").into()),
9            },
10            wgpu::ShaderRuntimeChecks::unchecked(),
11        )
12    }
13}

15struct Shaders {
16    ts: wgpu::ShaderModule,
17    ms: wgpu::ShaderModule,
18    fs: wgpu::ShaderModule,
19    ts_name: &'static str,
20    ms_name: &'static str,
21    fs_name: &'static str,
22}
23
24fn get_shaders(device: &wgpu::Device) -> Shaders {
25    let compiled = compile_wgsl(device);
26    Shaders {
27        ts: compiled.clone(),
28        ms: compiled.clone(),
29        fs: compiled.clone(),
30        ts_name: "ts_main",
31        ms_name: "ms_main",
32        fs_name: "fs_main",
33    }
34}

36pub struct Example {
37    pipeline: wgpu::RenderPipeline,
38}
39impl crate::framework::Example for Example {
40    fn init(
41        config: &wgpu::SurfaceConfiguration,
42        _adapter: &wgpu::Adapter,
43        device: &wgpu::Device,
44        _queue: &wgpu::Queue,
45    ) -> Self {
46        let Shaders {
47            ts,
48            ms,
49            fs,
50            ts_name,
51            ms_name,
52            fs_name,
53        } = get_shaders(device);
54        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
55            label: None,
56            bind_group_layouts: &[],
57            immediate_size: 0,
58        });
59        let pipeline = device.create_mesh_pipeline(&wgpu::MeshPipelineDescriptor {
60            label: None,
61            layout: Some(&pipeline_layout),
62            task: Some(wgpu::TaskState {
63                module: &ts,
64                entry_point: Some(ts_name),
65                compilation_options: Default::default(),
66            }),
67            mesh: wgpu::MeshState {
68                module: &ms,
69                entry_point: Some(ms_name),
70                compilation_options: Default::default(),
71            },
72            fragment: Some(wgpu::FragmentState {
73                module: &fs,
74                entry_point: Some(fs_name),
75                compilation_options: Default::default(),
76                targets: &[Some(config.view_formats[0].into())],
77            }),
78            primitive: wgpu::PrimitiveState {
79                cull_mode: Some(wgpu::Face::Back),
80                ..Default::default()
81            },
82            depth_stencil: None,
83            multisample: Default::default(),
84            multiview: None,
85            cache: None,
86        });
87        Self { pipeline }
88    }
89    fn render(&mut self, view: &wgpu::TextureView, device: &wgpu::Device, queue: &wgpu::Queue) {
90        let mut encoder =
91            device.create_command_encoder(&wgpu::CommandEncoderDescriptor { label: None });
92        {
93            let mut rpass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
94                label: None,
95                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
96                    view,
97                    resolve_target: None,
98                    ops: wgpu::Operations {
99                        load: wgpu::LoadOp::Clear(wgpu::Color {
100                            r: 0.1,
101                            g: 0.2,
102                            b: 0.3,
103                            a: 1.0,
104                        }),
105                        store: wgpu::StoreOp::Store,
106                    },
107                    depth_slice: None,
108                })],
109                depth_stencil_attachment: None,
110                timestamp_writes: None,
111                occlusion_query_set: None,
112                multiview_mask: None,
113            });
114            rpass.push_debug_group("Prepare data for draw.");
115            rpass.set_pipeline(&self.pipeline);
116            rpass.pop_debug_group();
117            rpass.insert_debug_marker("Draw!");
118            rpass.draw_mesh_tasks(1, 1, 1);
119        }
120        queue.submit(Some(encoder.finish()));
121    }
122    fn required_downlevel_capabilities() -> wgpu::DownlevelCapabilities {
123        Default::default()
124    }
125    fn required_features() -> wgpu::Features {
126        wgpu::Features::EXPERIMENTAL_MESH_SHADER | wgpu::Features::PASSTHROUGH_SHADERS
127    }
128    fn required_limits() -> wgpu::Limits {
129        wgpu::Limits::defaults().using_recommended_minimum_mesh_shader_values()
130    }
131    fn resize(
132        &mut self,
133        _config: &wgpu::SurfaceConfiguration,
134        _device: &wgpu::Device,
135        _queue: &wgpu::Queue,
136    ) {
137        // empty
138    }
139    fn update(&mut self, _event: winit::event::WindowEvent) {
140        // empty
141    }
142}

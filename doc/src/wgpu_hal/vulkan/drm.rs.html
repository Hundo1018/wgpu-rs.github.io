23macro_rules! to_u64 {
24    ($expr:expr) => {{
25        #[allow(trivial_numeric_casts)]
26        let expr = $expr as u64;
27        assert!(size_of_val(&expr) <= size_of::<u64>());
28        expr

33    /// Creates a new surface from the given drm fd and plane, deriving the connector and mode.
34    ///
35    /// # Safety
36    ///
37    /// - All parameters must point to valid DRM values.
38    pub fn create_surface_from_drm_plane(
39        &self,
40        fd: i32,
41        plane: u32,
42    ) -> Result<super::Surface, crate::InstanceError> {
43        let card = Card(fd);
44        let plane_info = card
45            .get_plane(
46                NonZeroU32::new(plane)
47                    .ok_or(crate::InstanceError::new("Invalid drm plane".to_string()))?
48                    .into(),
49            )
50            .map_err(|e| crate::InstanceError::with_source("drm plane not found".to_string(), e))?;
51        let crtc_handle = plane_info.crtc().ok_or(crate::InstanceError::new(
52            "No CRTC for drm plane".to_string(),
53        ))?;
54        let crtc_info = card
55            .get_crtc(crtc_handle)
56            .map_err(|e| crate::InstanceError::with_source("drm CRTC not found".to_string(), e))?;
57        let mode = crtc_info.mode().ok_or(crate::InstanceError::new(
58            "No mode for drm CRTC".to_string(),
59        ))?;
60
61        let mut connector_and_mode = None;
62        let resources = card.resource_handles().map_err(|e| {
63            crate::InstanceError::with_source("No drm resource handles found".to_string(), e)
64        })?;
65        for connector_handle in resources.connectors() {
66            let Ok(connector_info) = card.get_connector(*connector_handle, false) else {
67                continue;
68            };
69            if connector_info.state() != connector::State::Connected {
70                continue;
71            }
72
73            if let Some(encoder_handle) = connector_info.current_encoder() {
74                if let Ok(encoder_info) = card.get_encoder(encoder_handle) {
75                    if encoder_info.crtc() == Some(crtc_handle) {
76                        connector_and_mode = Some((*connector_handle, mode));
77                        break;
78                    }
79                }
80            }
81        }
82
83        let (connector_handle, mode) = connector_and_mode.ok_or(crate::InstanceError::new(
84            "Failed to derive drm connector and mode for plane".to_string(),
85        ))?;
86        let (width, height) = mode.size();
87        // Rate in millihertz
88        let refresh_rate = (((mode.clock() as f64 * 1000.0)
89            / (mode.hsync().2 as f64 * mode.vsync().2 as f64))
90            * 1000.0)
91            .round() as u32;
92        unsafe {
93            self.create_surface_from_drm(
94                fd,
95                plane,
96                connector_handle.into(),
97                width as u32,
98                height as u32,
99                refresh_rate,
100            )
101        }
102    }

104    /// Creates a new surface from the given drm configuration.
105    ///
106    /// # Safety
107    ///
108    /// - All parameters must point to valid DRM values.
109    pub unsafe fn create_surface_from_drm(
110        &self,
111        fd: i32,
112        plane: u32,
113        connector_id: u32,
114        width: u32,
115        height: u32,
116        refresh_rate: u32,
117    ) -> Result<super::Surface, crate::InstanceError> {
118        if !self
119            .shared
120            .extensions
121            .contains(&ext::acquire_drm_display::NAME)
122        {
123            return Err(crate::InstanceError::new(
124                "Vulkan driver does not support VK_EXT_acquire_drm_display".to_string(),
125            ));
126        }
127
128        let drm_stat = {
129            let mut stat = MaybeUninit::<libc::stat>::uninit();
130
131            if unsafe { libc::fstat(fd, stat.as_mut_ptr()) } != 0 {
132                return Err(crate::InstanceError::new(
133                    "Unable to fstat drm device".to_string(),
134                ));
135            }
136
137            unsafe { stat.assume_init() }
138        };
139
140        let raw_devices = match unsafe { self.shared.raw.enumerate_physical_devices() } {
141            Ok(devices) => devices,
142            Err(err) => {
143                log::error!("enumerate_adapters: {err}");
144                Vec::new()
145            }
146        };
147
148        let mut physical_device = None;
149
150        for device in raw_devices {
151            let properties2 = vk::PhysicalDeviceProperties2KHR::default();
152
153            let mut drm_props = vk::PhysicalDeviceDrmPropertiesEXT::default();
154            let mut properties2 = properties2.push_next(&mut drm_props);
155
156            unsafe {
157                self.shared
158                    .raw
159                    .get_physical_device_properties2(device, &mut properties2)
160            };
161
162            /*
163                The makedev call is just bit manipulation to combine major and minor device numbers into a Unix device ID.
164                It doesn't perform any filesystem operations, only bitshifting.
165                See: https://github.com/rust-lang/libc/blob/268e1b3810ac07ed637d9005bc1a54e49218c958/src/unix/linux_like/linux/mod.rs#L6049
166                We use the resulting device IDs to check if the Vulkan raw device from enumerate_physical_devices
167                matches the DRM device referred to by our file descriptor.
168            */
169
170            let primary_devid =
171                libc::makedev(drm_props.primary_major as _, drm_props.primary_minor as _);
172            let render_devid =
173                libc::makedev(drm_props.render_major as _, drm_props.render_minor as _);
174
175            // On most platforms, both `*_devid`s and `st_rdev` are `dev_t`s (which is generally
176            // observed to be an unsigned integral type no greater than 64 bits). However, on some
177            // platforms, there divergences from this pattern:
178            //
179            // - `armv7-linux-androideabi`: `dev_t` is `c_ulong`, and `*_devid`s are `dev_t`, but
180            //   `st_rdev` is `c_ulonglong`. So, we can't just do a `==` comparison.
181            // - OpenBSD has `dev_t` on both sides, but is `i32` (N.B., unsigned). Therefore, we
182            //   can't just use `u64::from`.
183            #[allow(clippy::useless_conversion)]
184            if [primary_devid, render_devid]
185                .map(|devid| to_u64!(devid))
186                .contains(&to_u64!(drm_stat.st_rdev))
187            {
188                physical_device = Some(device)
189            }
190        }
191
192        let physical_device = physical_device.ok_or(crate::InstanceError::new(
193            "Failed to find suitable drm device".to_string(),
194        ))?;
195
196        let acquire_drm_display_instance =
197            ext::acquire_drm_display::Instance::new(&self.shared.entry, &self.shared.raw);
198
199        let display = unsafe {
200            acquire_drm_display_instance
201                .get_drm_display(physical_device, fd, connector_id)
202                .expect("Failed to get drm display")
203        };
204
205        unsafe {
206            acquire_drm_display_instance
207                .acquire_drm_display(physical_device, fd, display)
208                .expect("Failed to acquire drm display")
209        }
210
211        let display_instance = khr::display::Instance::new(&self.shared.entry, &self.shared.raw);
212
213        let modes = unsafe {
214            display_instance
215                .get_display_mode_properties(physical_device, display)
216                .expect("Failed to get display modes")
217        };
218
219        let mut mode = None;
220
221        for current_mode in modes {
222            log::trace!(
223                "Comparing mode {}x{}@{} with {width}x{height}@{refresh_rate}",
224                current_mode.parameters.visible_region.width,
225                current_mode.parameters.visible_region.height,
226                current_mode.parameters.refresh_rate
227            );
228            if current_mode.parameters.refresh_rate == refresh_rate
229                && current_mode.parameters.visible_region.width == width
230                && current_mode.parameters.visible_region.height == height
231            {
232                mode = Some(current_mode)
233            }
234        }
235
236        let mode = mode.ok_or(crate::InstanceError::new(
237            "Failed to find suitable display mode".to_string(),
238        ))?;
239
240        let create_info = vk::DisplaySurfaceCreateInfoKHR::default()
241            .display_mode(mode.display_mode)
242            .image_extent(mode.parameters.visible_region)
243            .transform(vk::SurfaceTransformFlagsKHR::IDENTITY)
244            .alpha_mode(vk::DisplayPlaneAlphaFlagsKHR::OPAQUE)
245            .plane_index(plane);
246
247        let surface = unsafe { display_instance.create_display_plane_surface(&create_info, None) }
248            .expect("Failed to create DRM surface");
249
250        Ok(self.create_surface_from_vk_surface_khr(surface))
251    }
